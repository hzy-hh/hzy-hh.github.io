二叉树序列知二求一

决策树看的很烦，于是打算去刷一波二叉树的题。

参考自：https://blog.csdn.net/qq_44622401/article/details/104064901

#include <stdio.h>
#include <stdlib.h>
#include <ctype.h>
#include <stdbool.h>
#include <string.h>

void print_post(char *pre, char *in, int root, int start, int end);
void print_pre(char *post, char *in, int root, int start, int end);


int main()
{
    char *pre = "123456"; // 前序
    char *in = "324165";  // 中序
    char *post = "342651"; // 后序

    print_post(pre, in, 0, 0, strlen(in) -1 );
    printf("\n");
    print_pre(post, in, strlen(post)-1, 0, strlen(in)-1);
    printf("\n");
    system("pause");
    return 0;
}

void print_post( char *pre, char *in, int root, int start, int end)
{
    if( start > end )
        return;
    int i = start;
    while(i<end && in[i] != pre[root]) // 在中序序列中找到根节点以分割左右子树
        i++;
    print_post(pre, in, root+1, start, i - 1 ); // 遍历左子树
    print_post(pre, in, root+i-start+1, i+1, end); // 遍历右子树
    printf("%c ", pre[root]);    
}

void print_pre( char *post, char *in, int root, int start, int end)
{
    if( start>end )
        return;
    int i = start;
    while( i<end && in[i]!=post[root] )
        i++;
    printf("%c ", post[root]);
    print_pre( post, in, root-end+i-1, start, i-1); // 遍历左子树
    print_pre (post, in, root-1, i+1, end);
}
